    排序算法对比

    时间复杂度 空间复杂度 稳定性
    归并排序 nlogn n 稳定
    堆排序 nlogn 1 不稳定
    快速排序 nlogn logn 稳定

    插入排序 n^2 1 稳定
    冒泡排序 n^2 1 稳定
    选择排序 n^2 1 不稳定

    堆排序

    大顶堆要求父节点大于等于其2个子节点，小顶堆要求父节点小于等于其2个子节点。

    import numpy as np


def heapfix(arr, heapsize, index):
    root = index
    left = 2 * index
    right = 2 * index + 1
    if left < heapsize and arr[left-1] < arr[root-1]:
        root = left
    if right < heapsize and arr[right-1] < arr[root-1]:
        root = right
    if root != index:
        arr[index-1], arr[root-1] = arr[root-1], arr[index-1]
        heapfix(arr, heapsize, root)


def buildheap(arr, heapsize):
    index = heapsize/2
    while index > 0:
        heapfix(arr, heapsize,index)
        index -= 1


def heapsort(arr):
    heapsize = len(arr)
    buildheap(arr, heapsize)
    for i in range(heapsize):
        arr[0],arr[heapsize-1] = arr[heapsize-1], arr[0]
        heapsize -= 1
        heapfix(arr, heapsize,1)


if __name__ == "__main__":
    array = list(np.random.randint(0, 100, 10))
    heapsort(array)

    选择排序

    # -*- coding: utf-8 -*-
"""
选择排序：
1.从左至右遍历数组，找到最小的元素与第一个做交换；
2.在剩下的元素中继续从左至右遍历，并找出最小的与第二个元素做交换；
3.反复。
"""
import numpy as np


def select_sort(array):
    for i in range(len(array)):
        # tmp = array[i]
        min = array[i]
        min_index = i
        for j in range(i, len(array)):
            if array[j] < min:
                min = array[j]
                min_index = j
        array[min_index] = array[i]
        array[i] = min
    return array


if __name__ == '__main__':
    arr = np.random.randint(0, 100, 10)
    print '排序前：', arr
    arr_sort = select_sort(arr)
    print '排序后：', arr_sort

    插入排序

    # -*- coding: utf-8 -*-
"""
插入排序：
1.第一个元素认为已经排好序；
2.取下一个元素，和前面已经排好序的元素按从后前向的顺序进行比较，如果比前面的数小，就交换；如果比前面的数大，就停止；
3.重复第二步直到排序完成。
"""
import numpy as np


def insert_sort(arr):
    for i in range(1, len(arr)):
        for j in range(i, 0, -1):
            if arr[j-1] <= arr[j]:
                break
            else:
                tmp = arr[j]
                arr[j] = arr[j-1]
                arr[j-1] = tmp
    return arr


if __name__ == '__main__':
    array = np.random.randint(0, 100, 10)
    print '排序前：', array
    arr_sort = insert_sort(array)
    print '排序后：', arr_sort

    希尔排序

    # -*- coding: utf-8 -*-
"""
希尔排序：
1.设置步长d，将插入算法每次比较时前进的步长改为d;
2.减小d直到为1，重复第一步
"""
import numpy as np


def shell_sort(arr):
    d = len(arr)/2
    while d >= 1:
        for i in range(d, len(arr)):
            for j in range(i, 0, -d):
                if arr[j] > arr[j-d]:
                    break
                else:
                    tmp = arr[j]
                    arr[j] = arr[j-d]
                    arr[j-d] = tmp
        d = d/2
    return arr


if __name__ == '__main__':
    array = np.random.randint(0, 100, 10)
    print '排序前：', array
    arr_sort = shell_sort(array)
    print '排序后：', arr_sort

    归并排序

    # -*- coding: utf-8 -*-
"""
归并排序：
将序列不断的分为两个部分，直到每一部分只包含一个元素，进行归并
"""
import numpy as np


def merge_sort(arr):
    sort(arr, 0, len(arr)-1)
    return arr


def sort(arr, low, high):
    mid = (low + high) / 2
    if high > low:
        sort(arr, low, mid)
        sort(arr, mid + 1, high)
        merge(arr, low, mid, high)
    else:
        return


def merge(arr, low, mid, high):
    tmp = list(arr[low:high+1])
    i = low
    j = mid+1
    for k in range(low, high+1):
        if i > mid:
            arr[k] = tmp[j-low]
            j += 1
        elif j > high:
            arr[k] = tmp[i-low]
            i += 1
        elif tmp[i-low] < tmp[j-low]:
            arr[k] = tmp[i-low]
            i += 1
        else:
            arr[k] = tmp[j - low]
            j += 1


if __name__ == '__main__':
    array = np.random.randint(0, 100, 10)
    print '排序前：', array
    arr_sort = merge_sort(array)
    print '排序后：', arr_sort

    快速排序

    # -*- coding: utf-8 -*-
"""
快速排序：
1.选一个中轴数；
2.i，j分别从左向右、从右向左扫描，遇到比中轴数小的就放左边，大的就放右边；
3.对中轴数左右两边的元素继续1.2步骤，直到排序完成。
"""
import numpy as np


def quick_sort(arr):
    sort(arr, 0, len(arr)-1)
    return arr


def sort(arr, low, high):
    if low >= high:
        return arr
    else:
        mid = arr[low]
        i = low
        j = high
        while j >= i:
            if arr[i] > mid > arr[j]:
                tmp = arr[i]
                arr[i] = arr[j]
                arr[j] = tmp
            if arr[i] <= mid:
                i += 1
            if arr[j] >= mid:
                j -= 1
        arr[low] = arr[j]
        arr[j] = mid
        sort(arr, low, j-1)
        sort(arr, i, high)
    return arr


if __name__ == '__main__':
    array = np.random.randint(0, 100, 10)
    print '排序前：', array
    arr_sort = quick_sort(array)
    print '排序后：', arr_sort

    冒泡排序

    1. 比较相邻的元素。如果第一个比第二个大，就交换他们两个
    2. 对每一对相邻元素作同样的工作，从开始第一对到结尾的最后一对。在这一点，最后的元素应该会是最大的数。
    3. 针对所有的元素重复以上的步骤，除了最后一个。
    4. 持续每次对越来越少的元素重复上面的步骤，直到没有任何一对数字需要比较。
